import java.util.*;

public class MapFromLists {
    public static Map<String, String> mapNamesToValues(List<String> names, String pipeSeparated) {
        String[] values = pipeSeparated.split("\\|");
        Map<String, String> map = new HashMap<>();

        for (int i = 0; i < names.size() && i < values.length; i++) {
            map.put(names.get(i), values[i]);
        }
        return map;
    }

    public static void main(String[] args) {
        List<String> names = Arrays.asList("id", "firstName", "lastName");
        String data = "1|John|Doe";
        Map<String, String> result = mapNamesToValues(names, data);

        System.out.println(result); // {id=1, firstName=John, lastName=Doe}
    }
}
